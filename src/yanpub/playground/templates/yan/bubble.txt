def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n-i-1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
    return arr

nums = [5, 3, 8, 1, 2]
print("排序前：" + str(nums))
result = bubble_sort(nums[:])
print("排序后：" + str(result))
