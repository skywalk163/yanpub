#lang mingdao
# 明道 - 冒泡排序
# 经典排序算法：相邻元素两两比较，大的往后换

函数 冒泡排序(甲) {
  定义 乙 = 长度(甲)
  对于 丙 从 0 到 (乙 - 2) {
    对于 丁 从 0 到 (乙 - 丙 - 2) {
      如果 甲[丁] > 甲[丁 + 1] 那么 {
        定义 临时 = 甲[丁]
        甲[丁] = 甲[丁 + 1]
        甲[丁 + 1] = 临时
      }
    }
  }
  返回 甲
}

定义 数列 = [5, 3, 8, 1, 2, 9, 4, 7, 6]
打印 "排序前："
打印 数列
打印 "排序后："
打印 冒泡排序(数列)
