# 言律 - 图灵机原型机
# 模拟图灵机：二进制加一器

定 纸带 = ["1", "0", "1", "1", " "]
定 指针 = 0

定 状态表 = {}
状态表["S0 0"] = ["S1", "1", 0]
状态表["S0 1"] = ["S0", "0", -1]
状态表["S1 0"] = ["S1", "0", 0]
状态表["S1 1"] = ["S1", "1", 0]
状态表["S1  "] = ["S1", " ", 0]

定 当前状态 = "S0"
定 步数 = 0
输出("初始纸带：" + 纸带)

当 当前状态 != "停机" {
  定 读符 = 纸带[指针]
  定 键 = 当前状态 + " " + 读符
  如果 键 属于 状态表 那么 {
    定 转移 = 状态表[键]
    当前状态 = 转移[0]
    纸带[指针] = 转移[1]
    定 移动 = 转移[2]
    如果 移动 == 0 那么 { 当前状态 = "停机" }
    如果 移动 == -1 那么 { 指针 = 指针 - 1 }
    如果 移动 == 1 那么 { 指针 = 指针 + 1 }
  } 否则 {
    当前状态 = "停机"
  }
  步数 = 步数 + 1
  如果 步数 > 20 那么 {
    输出("超过最大步数，停止")
    当前状态 = "停机"
  }
}

输出("最终纸带：" + 纸带)
输出("执行步数：" + 步数)
